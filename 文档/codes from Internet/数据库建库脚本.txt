CREATE TABLE room (roomid integer (0) PRIMARY KEY UNIQUE NOT NULL);
CREATE TABLE adjust (roomid integer REFERENCES room (roomid) ON DELETE CASCADE ON UPDATE CASCADE MATCH SIMPLE NOT NULL, mode BOOLEAN (0), speed INTEGER (1, 3), starttemp INTEGER, stoptemp INTEGER, starttime DATETIME, stoptime DATETIME, fee DOUBLE);
CREATE TABLE openclose (roomid INTEGER (0) REFERENCES room (roomid) ON DELETE CASCADE ON UPDATE CASCADE MATCH SIMPLE NOT NULL, time DATETIME);
